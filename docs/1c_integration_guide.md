# Руководство по настройке 1С HTTP-сервиса
# Интеграция AIB OCR → 1С:Предприятие 8.3

## Обзор

AIB OCR отправляет данные в 1С через стандартный **HTTP-сервис** (Встроенный веб-сервис 1С).
Метод: `POST /hs/docai/documents/create`
Формат: `JSON` с заголовком `Authorization: Basic <base64>`

---

## Шаг 1: Создание HTTP-сервиса в конфигурации 1С

### 1.1 Через Конфигуратор

1. Откройте Конфигуратор → Конфигурация → Общие → **HTTP-сервисы**
2. Добавьте новый HTTP-сервис с именем `docai`
3. Корневой URL: `/docai`
4. Добавьте шаблон URL: `/documents/create` → метод `POST`

### 1.2 Код обработчика (встроенный язык 1С)

```bsl
// ─────────────────────────────────────────────────────────────
// HTTP-сервис: docai
// Шаблон URL: /documents/create
// Метод: POST
// ─────────────────────────────────────────────────────────────

Функция ОбработатьЗапрос(Запрос)
    
    // Чтение тела запроса
    ТелоЗапроса = Запрос.ПолучитьТелоКакСтроку("UTF-8");
    
    Если ТелоЗапроса = "" Тогда
        Возврат ОтветОшибка(400, "Пустое тело запроса");
    КонецЕсли;
    
    // Парсинг JSON
    ЧтениеJSON = Новый ЧтениеJSON;
    ЧтениеJSON.УстановитьСтроку(ТелоЗапроса);
    
    Попытка
        Данные = ПрочитатьJSON(ЧтениеJSON, Истина);
    Исключение
        Возврат ОтветОшибка(400, "Ошибка парсинга JSON: " + ОписаниеОшибки());
    КонецПопытки;
    
    // Извлечение полей из JSON
    ТипДокумента  = Данные["document_type"];
    ДанныеДок     = Данные["data"];
    
    // Создание документа в 1С в зависимости от типа
    Если ТипДокумента = "invoice" Тогда
        Результат = СоздатьСчетФактуру(ДанныеДок);
        
    ИначеЕсли ТипДокумента = "upd" Тогда
        Результат = СоздатьУПД(ДанныеДок);
        
    ИначеЕсли ТипДокумента = "torg12" Тогда
        Результат = СоздатьТоварнуюНакладную(ДанныеДок);
        
    Иначе
        Возврат ОтветОшибка(422, "Неизвестный тип документа: " + ТипДокумента);
    КонецЕсли;
    
    // Формируем ответ
    ОтветJSON = Новый ЗаписьJSON;
    ОтветJSON.УстановитьСтроку();
    ЗаписатьJSON(ОтветJSON, Результат);
    
    Ответ = Новый HTTPСервисОтвет(200);
    Ответ.УстановитьТелоИзСтроки(ОтветJSON.Закрыть(), "UTF-8");
    Ответ.Заголовки["Content-Type"] = "application/json; charset=utf-8";
    
    Возврат Ответ;
    
КонецФункции

// ─────────────────────────────────────────────────────────────
// Создание Счет-фактуры
// ─────────────────────────────────────────────────────────────
Функция СоздатьСчетФактуру(Данные)
    
    НовыйДок = Документы.СчетФактураПолученный.СоздатьДокумент();
    
    // Заполнение реквизитов
    НовыйДок.Дата   = ПарсироватьДату(Данные["document_date"]);
    НовыйДок.Номер  = Данные["document_number"];
    
    // Контрагент (поиск или создание)
    КонтрагентПоставщик = НайтиКонтрагентаПоИНН(Данные["supplier_inn"]);
    Если КонтрагентПоставщик <> Неопределено Тогда
        НовыйДок.Контрагент = КонтрагентПоставщик;
    КонецЕсли;
    
    // Суммы
    НовыйДок.СуммаДокумента    = Данные["total_amount"];
    НовыйДок.СуммаНДС           = Данные["vat_amount"];
    
    // Устанавливаем как черновик (требует проверки оператором)
    НовыйДок.ДополнительныеСвойства.Вставить("ИсточникOCR", Истина);
    НовыйДок.ДополнительныеСвойства.Вставить("УверенностьOCR",
        Данные["recognition_confidence"]);
    
    НовыйДок.Записать(РежимЗаписиДокумента.Запись);
    
    Возврат Новый Структура("success,document_ref,message",
        Истина,
        НовыйДок.Ссылка.УникальныйИдентификатор(),
        "Счет-фактура создана: " + НовыйДок.Ссылка);
        
КонецФункции

// ─────────────────────────────────────────────────────────────
// Вспомогательная: поиск контрагента по ИНН
// ─────────────────────────────────────────────────────────────
Функция НайтиКонтрагентаПоИНН(ИНН)
    
    Если НЕ ЗначениеЗаполнено(ИНН) Тогда
        Возврат Неопределено;
    КонецЕсли;
    
    Запрос = Новый Запрос;
    Запрос.Текст = 
        "ВЫБРАТЬ ПЕРВЫЕ 1
        |   Контрагенты.Ссылка
        |ИЗ
        |   Справочник.Контрагенты КАК Контрагенты
        |ГДЕ
        |   Контрагенты.ИНН = &ИНН
        |   И НЕ Контрагенты.ПометкаУдаления";
    
    Запрос.УстановитьПараметр("ИНН", ИНН);
    Результат = Запрос.Выполнить();
    
    Если Результат.Пустой() Тогда
        Возврат Неопределено;
    КонецЕсли;
    
    Возврат Результат.Выбрать().Следующий().Ссылка;
    
КонецФункции

// ─────────────────────────────────────────────────────────────
// Вспомогательная: парсинг даты ISO 8601 (YYYY-MM-DD)
// ─────────────────────────────────────────────────────────────
Функция ПарсироватьДату(СтрокаДата)
    
    Если НЕ ЗначениеЗаполнено(СтрокаДата) Тогда
        Возврат ТекущаяДата();
    КонецЕсли;
    
    // Формат: "2026-03-15"
    Год   = Число(Лев(СтрокаДата, 4));
    Месяц = Число(Сред(СтрокаДата, 6, 2));
    День  = Число(Прав(СтрокаДата, 2));
    
    Возврат Дата(Год, Месяц, День);
    
КонецФункции

// ─────────────────────────────────────────────────────────────
// Вспомогательная: ответ с ошибкой
// ─────────────────────────────────────────────────────────────
Функция ОтветОшибка(КодHTTP, Сообщение)
    
    Ответ = Новый HTTPСервисОтвет(КодHTTP);
    ТелоОтвета = "{""error"": """ + Сообщение + """}";
    Ответ.УстановитьТелоИзСтроки(ТелоОтвета, "UTF-8");
    Ответ.Заголовки["Content-Type"] = "application/json; charset=utf-8";
    Возврат Ответ;
    
КонецФункции
```

## Шаг 2: Публикация HTTP-сервиса на веб-сервере

### 2.1 Через Конфигуратор

1. **Администрирование → Публикация на веб-сервере...**
2. Имя публикации: `accounting`
3. Адрес веб-сервера: Apache или IIS
4. В разделе **HTTP-сервисы** — поставить галочку напротив `docai`
5. Нажать **Опубликовать**

Итоговый URL сервиса:
```
http://1c-server/accounting/hs/docai/documents/create
```

### 2.2 Настройка в .env AIB OCR

```env
C1_BASE_URL=http://1c-server/accounting/hs/docai
C1_USERNAME=docai_service
C1_PASSWORD=ваш_пароль_1с
```

---

## Шаг 3: Создание технической учётной записи в 1С

1. Откройте **Справочник → Пользователи**
2. Создайте пользователя `docai_service`
3. Установите пароль (совпадает с `C1_PASSWORD` в .env)
4. Роль: минимально необходимые права для создания документов
5. **Аутентификация 1С** — включена, **Аутентификация Windows** — выключена

## Решение проблем

| Проблема | Причина | Решение |
|----------|---------|---------|
| `401 Unauthorized` | Неверный логин/пароль | Проверить `C1_USERNAME` и `C1_PASSWORD` в .env |
| `404 Not Found` | Неверный URL | Проверить путь публикации и имя сервиса |
| `Connection refused` | 1С не запущен или неверный хост | Проверить `C1_BASE_URL` и доступность сети |
| Документ не создаётся | Ошибка в BSL коде | Смотреть технологический журнал 1С |
| Circuit Breaker OPEN | Много ошибок подряд | Ждать 60 сек или перезапустить worker |
